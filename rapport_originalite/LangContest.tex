% Exemple qui calcule la factorielle
\vspace{5mm}
\begin{lstlisting}
(fun (main) 
	(write (fact (read)))
)

(fun (fact x) (
	(if (= x 0) 
		(return 1)			
		(return (* x (fact (- x 1))))			
	)	
))
\end{lstlisting}

\begin{lstlisting}
// Construit une liste d'entiers
main()
{
	last := null;
	read (x);
	while (x != -1)
	{
		node := new/2;
		node.1 = last;
		node.2 := x;
		last = node;
		read (x);
	}
	y := sum (last);
	write (y);
}

// Calcule la somme des entiers d'une liste
sum (list)
{
	s := 0;
	current := list;
	while (current != null)
	{
		s := s + current.1;
		current := current.2;
	}
	return (s);
}
\end{lstlisting}
